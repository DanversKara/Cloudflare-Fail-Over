export default {
  async fetch(request) {

    const origin = "http://karas.page";

    try {

      const response = await fetch(origin, {
        method: "GET",
        headers: { "User-Agent": "Cloudflare-HealthCheck" },
        cf: { cacheTtl: 0 }
      });

      const text = await response.text();

      // detect nginx or error responses
      if (
        response.status >= 400 ||
        text.includes("404") ||
        text.includes("502") ||
        text.includes("Bad Gateway") ||
        text.includes("nginx")
      ) {
        throw new Error("Origin returned error page");
      }

      return new Response(text, response);

    } catch (error) {

      return fetch("https://karas-page-home.pages.dev");

    }

  }
}
